loadstring(game:HttpGet("https://raw.githubusercontent.com/ohmay5/Main/refs/heads/main/attachgun.txt"))()
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local Player = Players.LocalPlayer
local Net = ReplicatedStorage:WaitForChild("Modules"):WaitForChild("Net")
local RegisterAttack = Net:WaitForChild("RE/RegisterAttack")
local RegisterHit = Net:WaitForChild("RE/RegisterHit")

-- Cấu hình tốc độ "bình thường" hơn
local Settings = {
    Enabled = true,
    Cooldown = 0.4, -- Điều chỉnh thời gian giữa các lần đánh (giây) để tránh bị kick
}

local LastAttack = 0

local function GetTargets()
    local targets = {}
    local char = Player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return targets end
    
    local pos = char.HumanoidRootPart.Position
    local folder = workspace:FindFirstChild("Enemies") or workspace:FindFirstChild("Characters")
    
    if folder then
        for _, v in pairs(folder:GetChildren()) do
            if v ~= char and v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                if (v.HumanoidRootPart.Position - pos).Magnitude < 60 then
                    table.insert(targets, v)
                end
            end
        end
    end
    return targets
end

-- Chạy vòng lặp có kiểm soát thời gian
RunService.RenderStepped:Connect(function()
    if not Settings.Enabled then return end
    
    if tick() - LastAttack >= Settings.Cooldown then
        local targets = GetTargets()
        if #targets > 0 then
            local target = targets[1] -- Chỉ tập trung vào mục tiêu gần nhất để giống người chơi
            
            -- Gửi lệnh tấn công
            RegisterAttack:FireServer(0.5)
            RegisterHit:FireServer(target.Head, {{target, target.HumanoidRootPart}})
            
            LastAttack = tick()
        end
    end
end)
