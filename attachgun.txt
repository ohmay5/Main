local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Workspace = game:GetService("Workspace")

local Player = Players.LocalPlayer
local Net = ReplicatedStorage:WaitForChild("Modules"):WaitForChild("Net")
local ShootGunEvent = Net:WaitForChild("RE/ShootGunEvent")

-- CẤU HÌNH CHUNG
local Config = {
    MaxDistance = 100,
    MaxTargetsPerHit = 3,
}

-- CẤU HÌNH ĐỘ TRỄ RIÊNG CHO SÚNG
local GunSettings = {
    ["Dragonstorm"] = 0.15,  -- Súng Dragonstorm delay 0.8s
    ["Default"] = 0.2      -- Các loại súng khác mặc định delay 0.5s
}

local SmartFull = {}
SmartFull.__index = SmartFull

function SmartFull.new()
    return setmetatable({ LastGun = 0 }, SmartFull)
end

function SmartFull:GetTargets()
    local myPos = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") and Player.Character.HumanoidRootPart.Position
    if not myPos then return {} end

    local targets = {}  
    local function scan(folder)  
        if not folder then return end  
        for _, v in ipairs(folder:GetChildren()) do  
            local hum = v:FindFirstChild("Humanoid")  
            local hrp = v:FindFirstChild("HumanoidRootPart")  
            if v ~= Player.Character and hum and hum.Health > 0 and hrp and (hrp.Position - myPos).Magnitude <= Config.MaxDistance then  
                table.insert(targets, {Instance = v, Root = hrp, Dist = (hrp.Position - myPos).Magnitude})  
            end  
        end  
    end  
      
    scan(Workspace:FindFirstChild("Enemies"))  
    scan(Workspace:FindFirstChild("Characters"))  
    table.sort(targets, function(a, b) return a.Dist < b.Dist end)  
      
    local limited = {}  
    for i = 1, math.min(#targets, Config.MaxTargetsPerHit) do table.insert(limited, targets[i]) end  
    return limited
end

function SmartFull:Execute()
    local char = Player.Character
    local tool = char and char:FindFirstChildOfClass("Tool")
    if not tool then return end

    local targets = self:GetTargets()  
    if #targets == 0 then return end  
      
    local now = os.clock()  
    local tt = tool.ToolTip  

    -- CHỈ XỬ LÝ SÚNG (CÓ ĐỘ TRỄ RIÊNG TỪNG LOẠI)  
    if tt == "Gun" then  
        local delay = GunSettings[tool.Name] or GunSettings["Default"]
        
        if now - self.LastGun >= delay then  
            self.LastGun = now  
            local targetPos = targets[1].Root.Position  
            local targetModel = targets[1].Instance  
            ShootGunEvent:FireServer(targetPos, {targetModel:FindFirstChild("HumanoidRootPart")})  
        end  
    end
end

local Processor = SmartFull.new()
RunService.Heartbeat:Connect(function()
    pcall(function() Processor:Execute() end)
end)
