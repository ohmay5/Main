local function SaveSettings()

        writefile(

local function LoadSettings()

    if isfile and isfile(Path) then

        pcall(function()
            _G.SaveData =
            HttpService:JSONDecode(
                readfile(Path)
            )
        end)

    end

end


local function GetSetting(name,default)

    return _G.SaveData[name] ~= nil
    and _G.SaveData[name]
    or default

end

local FarmDistance = 325
local TweenSpeed = 250



local function TweenToEnemy(Mob)

    local Char=plr.Character
    if not Char then return end

    local Root =
    Char:FindFirstChild("HumanoidRootPart")

    local MobRoot =
    Mob:FindFirstChild("HumanoidRootPart")


    if not Root or not MobRoot then
        return
    end


    local Distance =
    (Root.Position-MobRoot.Position).Magnitude


    local Tween =
    TweenService:Create(

        Root,

        TweenInfo.new(
            Distance/TweenSpeed,
            Enum.EasingStyle.Linear
        ),

        {
        CFrame =
        MobRoot.CFrame*
        CFrame.new(0,15,0)
        }

    )


    Tween:Play()
    Tween.Completed:Wait()

end



Dungeon:AddToggle({

Name="Kill Mobs Nearest",

Default=
GetSetting(
"AutoFarmNear_Save",
false
),


Callback=function(v)

_G.AutoFarmNear=v

_G.SaveData.AutoFarmNear_Save=v

SaveSettings()

end

})
--========================
-- Auto Farm Loop
--========================

task.spawn(function()

while task.wait(0.3) do

    if _G.AutoFarmNear then

        pcall(function()

            local Char =
            plr.Character

            if not Char then return end


            local Root =
            Char:FindFirstChild(
            "HumanoidRootPart"
            )

            if not Root then return end


            local Folder =
            workspace:FindFirstChild(
            "Enemies"
            )


            local Target=nil
            local Distance=math.huge


            if Folder then

                for _,Mob in pairs(
                    Folder:GetChildren()
                ) do


                    local Hum =
                    Mob:FindFirstChild(
                    "Humanoid"
                    )


                    local MobRoot =
                    Mob:FindFirstChild(
                    "HumanoidRootPart"
                    )


                    if Hum
                    and MobRoot
                    and Hum.Health > 0 then


                        local Dis =
                        (
                        Root.Position -
                        MobRoot.Position
                        ).Magnitude


                        if Dis < Distance then

                            Distance=Dis
                            Target=Mob

                        end

                    end

                end

            end



            if Target
            and Distance <= FarmDistance then


                TweenToEnemy(Target)


                -- chờ đánh bằng hệ thống attack phần 3


                repeat task.wait(.2)

                until not _G.AutoFarmNear
                or not Target.Parent
                or (
                Target:FindFirstChild("Humanoid")
                and Target.Humanoid.Health<=0
                )


            end


        end)

    end

end

end)

local function GetCurrentFloor()

local Map =
workspace:FindFirstChild("Map")

if not Map then return end


local DungeonFolder =
Map:FindFirstChild("Dungeon")


if not DungeonFolder then return end


local Root =
HumanoidRootPart


local Closest=nil
local Dist=math.huge


for _,Floor in pairs(
DungeonFolder:GetChildren()
) do


local FloorRoot =
Floor:FindFirstChild("Root")


local Exit =
Floor:FindFirstChild(
"ExitTeleporter"
)


if FloorRoot and Exit then

local d =
(Root.Position-FloorRoot.Position)
.Magnitude


if d<Dist then

Dist=d
Closest=Floor

end


end


end


return Closest

end

end

end


end

local function BringMob(Mob)

local Root =
Mob:FindFirstChild(
"HumanoidRootPart"
)


local Hum =
Mob:FindFirstChild(
"Humanoid"
)


if not Root or not Hum then
return
end



local PlayerRoot =
HumanoidRootPart


if not PlayerRoot then return end



local Pos =
PlayerRoot.Position
+
PlayerRoot.CFrame.LookVector
*
BringOffset



Root.CFrame =
CFrame.new(Pos)


Root.CanCollide=false


Root.AssemblyLinearVelocity =
Vector3.zero


end




task.spawn(function()

while task.wait(.5) do


if _G.BringMobs then


pcall(function()


local Folder =
workspace:FindFirstChild(
"Enemies"
)


if Folder then


for _,Mob in pairs(
Folder:GetChildren()
) do


local Root =
Mob:FindFirstChild(
"HumanoidRootPart"
)


local Hum =
Mob:FindFirstChild(
"Humanoid"
)



if Root
and Hum
and Hum.Health>0 then


local Dis =
(
Root.Position -
HumanoidRootPart.Position
).Magnitude



if Dis <= BringDistance then

BringMob(Mob)

end


end


end


end


end)


end


end
